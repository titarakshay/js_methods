function map(square, numArr) {
  let newArray = [];
  
  for(let i = 0; i < numArr.length; i++) {
    let returnedValue = square(numArr[i], i, numArr);
    newArray.push(returnedValue);
  }

  return newArray;
}

function square(val) {
  return (val ** 2);
